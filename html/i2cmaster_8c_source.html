00032   Issues a start condition and sends address and transfer direction.
00033   return 0 = device accessible, 1= failed to access device
00034 *************************************************************************/
00035 unsigned char i2c_start(unsigned char address)
00036 {
00037     uint8_t   twst;
00038 
00039         // send START condition
00040         TWCR = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
00041 
00042         // wait until transmission completed
00043         while(!(TWCR & (1<<TWINT)));
00044 
00045         // check value of TWI Status Register. Mask prescaler bits.
00046         twst = TW_STATUS & 0xF8;
00047         if ( (twst != TW_START) && (twst != TW_REP_START)) return 1;
00048 
00049         // send device address
00050         TWDR = address;
00051         TWCR = (1<<TWINT) | (1<<TWEN);
00052 
00053         // wail until transmission completed and ACK/NACK has been received
00054         while(!(TWCR & (1<<TWINT)));
00055 
00056         // check value of TWI Status Register. Mask prescaler bits.
00057         twst = TW_STATUS & 0xF8;
00058         if ( (twst != TW_MT_SLA_ACK) && (twst != TW_MR_SLA_ACK) ) return 1;
00059 
00060         return 0;
00061 
00062 }/* i2c_start */
